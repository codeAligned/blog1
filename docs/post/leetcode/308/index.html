typedef long long ll;

struct BIT2D {
    vector<vector<int>> bit;
    int n, m;
    void init(int _n, int _m)
    {
        n = _n + 3;
        m = _m + 3;
        bit = vector<vector<int>>(n, vector<int>(m, 0));
    }

    ll sum(int a, int b)
    {
        a += 2;
        b += 2;
        ll s = 0;
        for (int i = a; i > 0; i -= (i & -i))
            for (int j = b; j > 0; j -= (j & -j))
                s += bit[i][j];
        return s;
    }

    void add(int a, int b, ll x)
    {
        a += 2;
        b += 2;
        for (int i = a; i < n; i += (i & -i))
            for (int j = b; j < m; j += (j & -j))
                bit[i][j] += x;
    }
} bit;

class NumMatrix
{
private:
    vector<vector<int>> matrix;

public:
    NumMatrix(vector<vector<int>> _matrix)
    {
        matrix = _matrix;

        int n = matrix.size();
        if (n == 0)
            return;
        int m = matrix[0].size();
        bit.init(n, m);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                bit.add(i, j, matrix[i][j]);
            }
        }
    }

    void update(int row, int col, int val)
    {
        int diff = val - matrix[row][col];
        bit.add(row, col, diff);
        matrix[row][col] = val;
    }

    int sumRegion(int row1, int col1, int row2, int col2)
    {
        int ret = bit.sum(row2, col2) - bit.sum(row1 - 1, col2) -
                  bit.sum(row2, col1 - 1) + bit.sum(row1 - 1, col1 - 1);
        return ret;
    }
};
