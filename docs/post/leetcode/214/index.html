// handle special cases first
// [], "", ...
// range of input?
class Solution
{
private:
    bool isPalindrome(string s)
    {
        int n = (int)s.length();
        for (int i = 0; i < n / 2; i++) {
            if (s[i] != s[n - 1 - i])
                return false;
        }
        return true;
    }

public:
    string shortestPalindrome(string s)
    {
        for (int i = (int)s.length(); i >= 1; i--) {
            // [0, i]
            if (isPalindrome(s.substr(0, i))) {
                string front = s.substr(i);
                reverse(front.begin(), front.end());
                return front + s;
            }
        }

        return s;
    }
};
