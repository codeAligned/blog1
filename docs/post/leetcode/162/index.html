    int findPeakElement(vector<int> &nums)
    {
        int n = (int)nums.size(), l = 0, r = n - 1;
        while (r - l > 1) {
            int mid = l + (r - l) / 2;

            if (nums[mid] < nums[mid + 1])
                l = mid;
            else
                r = mid;
        }

        return nums[r] > nums[l] ? r : l;
    }
